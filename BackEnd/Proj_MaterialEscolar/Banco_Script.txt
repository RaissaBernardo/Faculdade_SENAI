CREATE DATABASE IF NOT EXISTS materialescolar;
USE materialescolar;

CREATE TABLE Cliente (
    cpf VARCHAR(14) PRIMARY KEY,
    nome VARCHAR(100),
    telefone VARCHAR(20),
    email VARCHAR(100)
);

CREATE TABLE Endereco (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_cliente VARCHAR(14),
    rua VARCHAR(150),
    cep VARCHAR(20),
    num VARCHAR(10),
    bairro VARCHAR(100),
    complemento VARCHAR(100),
    FOREIGN KEY (id_cliente) REFERENCES Cliente(cpf)
);


CREATE TABLE Produto (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tamanho VARCHAR(20),
    cor VARCHAR(30),
    marca VARCHAR(50),
    genero VARCHAR(20),
    preco DECIMAL(10,2),
    estoque INT,
    descricao VARCHAR(255)
);

CREATE TABLE Compra (
    idCompra INT AUTO_INCREMENT PRIMARY KEY,
    data_compra DATE,
    data_entrega DATE,
    id_produto INT,
    id_cliente VARCHAR(14),
    quantidade INT,
    valorTotal DECIMAL(10,2),
    FOREIGN KEY (id_produto) REFERENCES Produto(id),
    FOREIGN KEY (id_cliente) REFERENCES Cliente(cpf)
);
